Variant la intensitat d'un led:Utilitzant pulsadors i PWN de l'arduino

Podem veure una simulació, hem creat ua variable step que té un valor de 17, multiple de 255. Depenent el botó que apretem,el primer o el segon, pujarà o baixarà aquesta quantitat.

següent enllaç 


         
// ────────────────────────────────────────────────
// VARIABLES GLOBALS (visibles a tot el programa)
// ────────────────────────────────────────────────
const int pinLed       = 11;     // Pin PWM (\~) on connectem el LED
const int upButton     = 2;      // Botó per AUGMENTAR intensitat
const int downButton   = 3;      // Botó per REDUIR intensitat

int  brightness = 0;             // Valor actual (0 = apagat, 255 = màxim)
const int step      = 17;        // Canvi per cada pulsació (255 ÷ 17 ≈ 15 passos)

bool lastUpState   = HIGH;       // Últim estat conegut del botó pujar
bool lastDownState = HIGH;       // Últim estat conegut del botó baixar

// ────────────────────────────────────────────────
// SETUP – s’executa només un cop quan engeguem l’Arduino
// ────────────────────────────────────────────────
void setup() {
  Serial.begin(115200);          // Per veure la brillantor pel monitor sèrie
  Serial.println("Control LED amb botons iniciat");

  pinMode(LED_BUILTIN, OUTPUT);  // LED integrat a la placa (per debug)
  pinMode(pinLed,      OUTPUT);  // El nostre LED extern

  // Els botons utilitzen resistència interna PULL-UP
  pinMode(upButton,   INPUT_PULLUP);
  pinMode(downButton, INPUT_PULLUP);
}

// ────────────────────────────────────────────────
// LOOP – s’executa infinitament
// ────────────────────────────────────────────────
void loop() {
  // Llegim l’estat actual dels botons
  bool upState   = digitalRead(upButton);
  bool downState = digitalRead(downButton);

  // Detectem caiguda de flanc (HIGH → LOW) → botó acabat de prémer
  if (lastUpState == HIGH && upState == LOW) {
    brightness += step;
    if (brightness > 255) brightness = 255;   // Límit superior
    delay(200);                                // Anti-rebot simple
  }

  if (lastDownState == HIGH && downState == LOW) {
    brightness -= step;
    if (brightness < 0) brightness = 0;       // Límit inferior
    delay(200);                                // Anti-rebot simple
  }

  // Actualitzem l’estat anterior per la següent volta
  lastUpState   = upState;
  lastDownState = downState;

  // Enviem el valor PWM al LED (0 → 255)
  analogWrite(pinLed, brightness);

  // Mostrem el valor actual al monitor sèrie
  Serial.println(brightness);

  // Petit retard per no saturar el port sèrie
  delay(200);

  // ── Opció debug: parpelleig lent del LED integrat ──
  //digitalWrite(LED_BUILTIN, HIGH);
  //delay(1000);
  //digitalWrite(LED_BUILTIN, LOW);
  //delay(1000);
}
